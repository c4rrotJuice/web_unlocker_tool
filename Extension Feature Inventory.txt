  tier: Tier;
  status: "active" | "expired" | "grace_period";

  capabilities: {
    // --- PAGE UTILITY ---
    copy_unlock: {
      enabled: boolean;        // restore copy/select/paste
      right_click_restore: boolean;
      anti_block_bypass: boolean;
      soft_ad_cleanup: boolean;
    };

    // --- CAPTURE ACTIONS ---
    actions: {
      copy: boolean;
      cite: boolean;
      note: boolean;
      quote: boolean;
      work_in_editor: boolean;
    };

    // --- USAGE LIMITS ---
    usage: {
      citations_per_week: number | null;
      notes_per_week: number | null;
      documents_limit: number | null;
      history_limit: number | "limited" | "full";
    };

    // --- CITATION SYSTEM ---
    citation: {
      styles: Array<"apa" | "mla" | "chicago" | "harvard">;
      formats: Array<"inline" | "footnote" | "bibliography">;
      backend_render_only: true; // enforced always 
    };

    // --- RESEARCH GRAPH ACCESS ---
    research: {
      sources: boolean;
      citations: boolean;
      quotes: boolean;
      notes: boolean;
      documents: boolean;
      projects: boolean;
      tags: boolean;
    };

    // --- EDITOR ---
    editor: {
      access: boolean;
      autosave: boolean;
      checkpoints: boolean;
      export_formats: Array<"pdf" | "html">;
      zip_export: boolean;
      document_freeze_on_limit: boolean;
    };

    // --- HISTORY / CONTINUITY ---
    history: {
      recent_items: boolean;
      extended_history: boolean;
      search: boolean;
      export: boolean;
    };

    // --- EXTENSION BEHAVIOR ---
    extension: {
      pill_actions: {
        copy: boolean;
        cite: boolean;
        note: boolean;
        quote: boolean;
      };
      sidepanel_access: boolean;
      work_in_editor_flow: boolean;
    };

    // --- MONETIZATION FLAGS ---
    monetization: {
      ads: boolean;
      upgrade_prompts: boolean;
    };
  };
};
